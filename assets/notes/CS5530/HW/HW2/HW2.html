CREATE TABLE Games (
	round_number int unsigned NOT NULL AUTO_INCREMENT,
	name varchar(255) NOT NULL,
	site varchar(255) NOT NULL,
	date date NOT NULL,
	player_b varchar(255) NOT NULL,
	player_w varchar(255) NOT NULL,
	result char(1) NOT NULL,
	list_of_moves varchar(1000) NOT NULL,
	PRIMARY KEY
	(round_number, name, site, date, player_b, player_w, result, list_of_moves),
	FOREIGN KEY(name, site, date) REFERENCES Events(name, site, date),
	FOREIGN KEY(player_b) REFERENCES Players(name),
	FOREIGN KEY(player_w) REFERENCES Players(name)
);
